//REGION Check if any players are in the path of the boulder
//We are purposefully setting this up so that the goblin AI seems stupid. 
//They will gladly kill their own if it means killing a player

IF
EnteredTrigger(_Player,S_GOB_Checkpoint_AiBoulderPath_Trigger_c9a25ff6-b0ad-436b-a468-dd56911b867c)
THEN
PROC_GOB_Checkpoint_CheckPlayersInBouldersPath();

IF
CombatRoundStarted(_CmbId,_)
AND
DB_GOB_Checkpoint_CanTargetRollingBoulder(_Goblin)
AND
DB_Is_InCombat(_Goblin,_CmbId)
AND
IsDestroyed(S_GOB_Checkpoint_CMB_BoulderRope_3705c29b-50fb-405d-a197-bb4f01253ab2,0)
THEN
PROC_GOB_Checkpoint_CheckPlayersInBouldersPath();

PROC
PROC_GOB_Checkpoint_CheckPlayersInBouldersPath()
AND
IsDestroyed(S_GOB_Checkpoint_CMB_BoulderRope_3705c29b-50fb-405d-a197-bb4f01253ab2,0)
AND
QRY_GOB_Checkpoint_PlayersInBouldersPath()
AND
DB_GOB_Checkpoint_PlayersInBouldersPath(1)
AND
DB_GOB_Checkpoint_CanTargetRollingBoulder(_Goblin)
AND
DB_Is_InCombat((CHARACTER)_Goblin,_CmbId)
AND
QRY_CurrentCombatRoundIsGreaterThan(_CmbId,1)
THEN
AiAddInterestingItem(_Goblin,S_GOB_Checkpoint_CMB_BoulderRope_3705c29b-50fb-405d-a197-bb4f01253ab2);

IF
LeftTrigger(_Player,S_GOB_Checkpoint_AiBoulderPath_Trigger_c9a25ff6-b0ad-436b-a468-dd56911b867c)
AND
IsDestroyed(S_GOB_Checkpoint_CMB_BoulderRope_3705c29b-50fb-405d-a197-bb4f01253ab2,0)
AND
QRY_GOB_Checkpoint_PlayersInBouldersPath()
AND
NOT DB_GOB_Checkpoint_PlayersInBouldersPath(1)
AND
DB_GOB_Checkpoint_CanTargetRollingBoulder(_Goblin)
THEN
AiRemoveInterestingItem((CHARACTER)_Goblin,S_GOB_Checkpoint_CMB_BoulderRope_3705c29b-50fb-405d-a197-bb4f01253ab2);

IF
DestroyedBy(S_GOB_Checkpoint_CMB_BoulderRope_3705c29b-50fb-405d-a197-bb4f01253ab2,_,_,_)
AND
DB_GOB_Checkpoint_CanTargetRollingBoulder(_Goblin)
THEN
AiRemoveInterestingItem((CHARACTER)_Goblin,S_GOB_Checkpoint_CMB_BoulderRope_3705c29b-50fb-405d-a197-bb4f01253ab2);


//Goblins dont care if there are friendlies in the path of the boulder
QRY
QRY_GOB_Checkpoint_PlayersInBouldersPath()
THEN
NOT DB_GOB_Checkpoint_PlayersInBouldersPath(1);

QRY
QRY_GOB_Checkpoint_PlayersInBouldersPath()
AND
DB_InRegion(_Player,S_GOB_Checkpoint_AiBoulderPath_Trigger_c9a25ff6-b0ad-436b-a468-dd56911b867c)
AND
DB_Players(_Player)
AND
NOT DB_GOB_Checkpoint_PlayersInBouldersPath(1)
THEN
DB_GOB_Checkpoint_PlayersInBouldersPath(1);


//END_REGION
