//REGION Bloodless Boars
IF
DB_ORI_AstarionHunger_Boars(_Boar, _)
THEN
PROC_SetOnStage(_Boar, 0);

// Getting Ceremorphosis 2 night OR entering the forest area (which cancels Ceremorphosis 2 night) does the following:
// a) reveals the boars
// b) unlocks the bite scene
// if the night is cancelled due to us entering the forest, then we only reveal the boars
// if Astarion took at least one long rest with us (he couldn't hunt an animal if we never had a long rest after recruiting him)
IF
FlagSet((FLAG)NIGHT_Ceremorphosis2_485f589f-0da1-4eea-9a28-210f91837a69, _, _)
AND
NOT DB_Avatars(S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255)
THEN
PROC_GlobalSetFlagAndCache(ORI_Astarion_State_BiteSceneUnlocked_3cfdb2af-c0b1-4820-a635-cf1e5afbf0ee);
PROC_ORI_Astarion_ShowBoars();

PROC
PROC_LongRest()
AND
DB_PartOfTheTeam((CHARACTER)S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255)
AND
NOT DB_ORI_Astarion_TookAtLeastOneLongRest(1)
THEN
DB_ORI_Astarion_TookAtLeastOneLongRest(1);

IF
FlagSet(FOR_PartyProgress_EnteredForest_96c8ee70-eba5-4007-855f-59bef6c9f051, _, _)
AND
NOT DB_Avatars(S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255)
AND
NOT DB_GlobalFlag((FLAG)ORI_Astarion_State_BiteSceneUnlocked_3cfdb2af-c0b1-4820-a635-cf1e5afbf0ee)
THEN
PROC_GlobalSetFlagAndCache(ORI_Astarion_State_BiteSceneUnlocked_3cfdb2af-c0b1-4820-a635-cf1e5afbf0ee);
PROC_ORI_Astarion_CheckShowBoars();

PROC
PROC_ORI_Astarion_CheckShowBoars()
AND
DB_ORI_Astarion_TookAtLeastOneLongRest(1)
THEN
PROC_ORI_Astarion_ShowBoars();

// For Astarion Avatar we need to show the boars if we had a Bite Scene
// but instead of biting a companion decided to go and hunt an animal
IF
FlagSet((FLAG)CAMP_Astarion_TheHungerA_HuntAnimal_ef9c902d-0940-4a39-a926-5717795b89fe, _, _)
THEN
PROC_ORI_Astarion_ShowBoars();

// if we reveal the boars in the real time, let's make sure
// that there are at least no players around
PROC
PROC_ORI_Astarion_ShowBoars()
AND
QRY_OnlyOnce("ORI_Astarion_ShowBoars")
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
GetClosestPlayer(_Boar, _, _Dist)
AND
_Dist > 30.0
THEN
PROC_SetOnStage(_Boar, 1);
PROC_TriggerRegisterForPlayers(_BoarTrigger);
TriggerRegisterForCharacter(_BoarTrigger, _Boar);

// When one of the boars is discovered we disable the rest of them:
// if there is no player nearby then we hide them immediately
// otherwise we will hide them after a long rest
IF
EnteredTrigger(_Player, _BoarTrigger)
AND
DB_Players(_Player)
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
NOT DB_GlobalFlag(ORI_AstarionHunger_State_BoarDiscovered_a275c3d9-7ffc-4f33-85ea-e15227534dc3)
THEN
DB_ORI_AstarionHunger_Boars_DiscoveredBoar(_Boar);
PROC_ORI_Astarion_Boars_HideUndiscoveredBoars();
PROC_GlobalSetFlagAndCache(ORI_AstarionHunger_State_BoarDiscovered_a275c3d9-7ffc-4f33-85ea-e15227534dc3);

// When a non-Astarion player enters the trigger we also play a voicebark
IF
EnteredTrigger(_Player, _BoarTrigger)
AND
DB_Players(_Player)
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
_Player != S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255
AND
QRY_OnlyOnce("OR_Astarion_Boars_DiscoveredBoarVB")
THEN
StartVoiceBark((VOICEBARKRESOURCE)ORI_AstarionHunger_VB_BoarFound_a8dc0448-c51e-f3d6-6f44-74767f1a7c32, _Player);

PROC
PROC_ORI_Astarion_Boars_HideUndiscoveredBoars()
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
NOT DB_ORI_AstarionHunger_Boars_DiscoveredBoar(_Boar)
THEN
PROC_TriggerUnregisterForPlayers(_BoarTrigger);

// if no Player near the boar, we simply remove the boar
// otherwise it will be removed after a long rest
PROC
PROC_ORI_Astarion_Boars_HideUndiscoveredBoars()
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
NOT DB_ORI_AstarionHunger_Boars_DiscoveredBoar(_Boar)
AND
GetClosestPlayer(_Boar, _Player, _Dist)
AND
_Dist >= 30.0
THEN
PROC_SetOnStage(_Boar, 0);

PROC
PROC_ORI_Astarion_Boars_HideUndiscoveredBoars()
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
NOT DB_ORI_AstarionHunger_Boars_DiscoveredBoar(_Boar)
AND
GetClosestPlayer(_Boar, _Player, _Dist)
AND
_Dist < 30.0
THEN
DB_ORI_AstarionHunger_Boars_RemoveAfterLongRest(_Boar);

// if non-Astarion loots the boar and Astarion is nearby
// we start ORI_AstarionHunger_BoarExamine_POM_Astarion dialog with the character that looted the body and Astarion (only once)
// unless the character failed to examine the boar before
// however, if Astarion is Avatar, then we don't trigger an interactive dialog. It's just a corpse.
IF
CharacterLootedCharacter(_Player, _Boar)
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
DB_ORI_AstarionHunger_Boars_DiscoveredBoar(_Boar)
AND
NOT DB_GlobalFlag((FLAG)CAMP_Astarion_TheHungerA_HuntAnimal_ef9c902d-0940-4a39-a926-5717795b89fe) 
AND
NOT DB_OnlyOnce("ORI_AstarionHunger_Boars_DiscussedWithAstarion")
AND
DB_Players(_Player)
AND
_Player != S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255
AND
GetFlag((FLAG)ORI_Astarion_BoarInvestigation_State_Failure_afff5e2d-da09-4cab-84d6-16fc99fd8f72, _Player, 0) // failed to examine before - no OM!
AND
GetDistanceTo(_Player, S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255, _Dist)
AND
_Dist < 10.0
AND
IsInInventory(_Boar, 0)
AND
QRY_StartDialog_Fixed(ORI_AstarionHunger_BoarExamine_POM_Astarion_9b32382e-53a9-d78d-4e16-ce30daf584fc, S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255, _Player, _Boar)
THEN
DB_OnlyOnce("ORI_AstarionHunger_Boars_DiscussedWithAstarion");

// if non-Astarion loots the boar and Astarion is NOT nearby
// then we start ORI_AstarionHunger_BoarExamine dialog, but only if the player hasn't actually tried to examine the boar yet
IF
CharacterLootedCharacter(_Player, _Boar)
AND
DB_ORI_AstarionHunger_Boars(_Boar, _BoarTrigger)
AND
DB_ORI_AstarionHunger_Boars_DiscoveredBoar(_Boar)
AND
NOT DB_GlobalFlag((FLAG)CAMP_Astarion_TheHungerA_HuntAnimal_ef9c902d-0940-4a39-a926-5717795b89fe) 
AND
NOT DB_OnlyOnce("ORI_AstarionHunger_Boars_DiscussedWithAstarion")
AND
DB_Players(_Player)
AND
_Player != S_Player_Astarion_c7c13742-bacd-460a-8f65-f864fe41f255
AND
GetFlag((FLAG)ORI_Astarion_BoarInvestigation_State_Attempted_f73bce2c-f898-4f93-860a-f1ea8c1d1cbd, _Player, 0)
AND
IsInInventory(_Boar, 0)
AND
QRY_StartDialog_Fixed((DIALOGRESOURCE)ORI_AstarionHunger_BoarExamine_2fa501df-bef4-31ed-9ff0-07b08a4474f5, _Player, _Boar)
THEN
DB_NOOP(1);

// if the player moves a boar outside the initial trigger (that includes picking up the corpse)
// we don't remove it after a long rest
IF
LeftTrigger(_Boar, _BoarTrigger)
AND
DB_ORI_AstarionHunger_Boars_RemoveAfterLongRest(_Boar)
THEN
NOT DB_ORI_AstarionHunger_Boars_RemoveAfterLongRest(_Boar);

NOT DB_OnlyOnce("ORI_AstarionHunger_Boars_DiscussedWithAstarion")
