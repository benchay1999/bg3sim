//REGION Toxic wall
PROC
PROC_UND_ToxicWall_On((CHARACTER)_User, (ITEM)_ToxicWall)
THEN
SetDualEntityEvent(_User, _ToxicWall,"UND_ToxicWall_On", 1);
SetFlag(UND_MyconidCircle_State_WallActive_c0d66b03-cec5-4ac1-915b-8c612c8d1594, _ToxicWall, 0);

PROC
PROC_UND_ToxicWall_Off((CHARACTER)_User, (ITEM)_ToxicWall)
THEN
SetDualEntityEvent(_User, _ToxicWall,"UND_ToxicWall_Off", 1);
ClearFlag(UND_MyconidCircle_State_WallActive_c0d66b03-cec5-4ac1-915b-8c612c8d1594, _ToxicWall, 0);

PROC
PROC_UND_ToxicWall_Switch((CHARACTER)_User, (ITEM)_ToxicWall)
AND
NOT QRY_UND_ToxicWall_DeactivateWallIfActive(_User, _ToxicWall)
THEN
PROC_UND_ToxicWall_On(_User, _ToxicWall);

AND

THEN
SetEntityEvent(_Myconid, "UND_MyconidCircle_DeactivateWall", 1);

QRY
