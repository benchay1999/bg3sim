//REGION Arrest
IF
DB_TWN_MarchingArmy_PrisonTrigger(_PrisonTrigger)
THEN
PROC_TriggerRegisterForPlayers(_PrisonTrigger);

IF
EnteredTrigger(_Player, S_TWN_MarchingArmy_ArrestTrigger_f17b1dd9-746f-48a5-8864-eb41f8c3f58f)
AND
DB_Players(_Player)
THEN
PROC_TWN_MarchingArmy_StartArrestDialog(_Player);

PROC
PROC_TWN_MarchingArmy_StartArrestDialog((CHARACTER)_Player)
AND
NOT QRY_TWN_MarchingArmy_TryStartArrestDialog(_Player)
AND
NOT DB_TWN_MarchingArmy_MainCriminal(_)
THEN
DB_TWN_MarchingArmy_MainCriminal((CHARACTER)_Player);
PROC_TWN_MarchingArmy_PrepareCriminals(_Player);
PROC_TWN_MarchingArmy_TeleportAllCriminalsToPrison(_Player);

QRY
QRY_TWN_MarchingArmy_TryStartArrestDialog((CHARACTER)_Player)
AND
NOT DB_TWN_MarchingArmy_MainCriminal(_)
AND
QRY_StartDialog(TWN_MarchingArmy_Arrest_17c8a1e6-2855-601f-1dfd-535352e93fd4, _Player)
THEN
DB_TWN_MarchingArmy_MainCriminal((CHARACTER)_Player);

// dialog success & player backed off - teleport away
IF
DialogEnded(TWN_MarchingArmy_Arrest_17c8a1e6-2855-601f-1dfd-535352e93fd4, _ID)
AND
DB_TWN_MarchingArmy_MainCriminal(_Player)
AND
DB_DialogPlayers(_ID, _Player, _)
AND
GetFlag(TWN_MarchingArmy_Event_BackedOff_7cdfe338-196e-4e9d-88b0-901684611175, _Player, 1)
THEN
PROC_TWN_MarchingArmy_PrepareCriminals(_Player);
PROC_TWN_MarchingArmy_TeleportCriminalsAway(_Player);

IF
DialogEnded(TWN_MarchingArmy_Arrest_17c8a1e6-2855-601f-1dfd-535352e93fd4, _ID)
AND
DB_TWN_MarchingArmy_MainCriminal(_Player)
AND
DB_DialogPlayers(_ID, _Player, _)
AND
GetFlag(TWN_MarchingArmy_Event_BackedOff_7cdfe338-196e-4e9d-88b0-901684611175, _Player, 0)
THEN
PROC_TWN_MarchingArmy_PrepareCriminals(_Player);
PROC_TWN_MarchingArmy_TeleportAllCriminalsToPrison(_Player);

PROC
PROC_TWN_MarchingArmy_PrepareCriminals((CHARACTER)_MainCriminal)
AND
DB_Players(_OtherCriminal) // including the main criminal (original _Player)
AND
IsInPartyWith(_OtherCriminal, _MainCriminal, 1)
AND
InSamePartyGroup(_OtherCriminal, _MainCriminal, 1)
THEN
DB_TWN_MarchingArmy_Criminals((CHARACTER)_MainCriminal, (CHARACTER)_OtherCriminal); // need a separate DB. Teleporting the first criminal in prison unlinks him from the group

PROC
PROC_TWN_MarchingArmy_TeleportCriminalsAway((CHARACTER)_MainCriminal)
AND
DB_TWN_MarchingArmy_Criminals(_MainCriminal, _OtherCriminal)
THEN
PROC_ForceStopDialog(_OtherCriminal);
TeleportTo(_OtherCriminal, S_TWN_MarchingArmy_OutTrigger_8125fba4-005f-4096-8fe8-eb4f25e086fd, "", 0);
ClearFlag(TWN_MarchingArmy_Event_BackedOff_7cdfe338-196e-4e9d-88b0-901684611175, _MainCriminal);
NOT DB_TWN_MarchingArmy_MainCriminal(_MainCriminal);

PROC
PROC_TWN_MarchingArmy_TeleportAllCriminalsToPrison((CHARACTER)_MainCriminal)
AND
DB_TWN_MarchingArmy_Criminals(_MainCriminal, _OtherCriminal)
THEN
PROC_ForceStopDialog(_OtherCriminal);
PROC_TWN_MarchingArmy_TeleportCriminalToPrison((CHARACTER)_OtherCriminal);

PROC
PROC_TWN_MarchingArmy_TeleportAllCriminalsToPrison((CHARACTER)_MainCriminal)
AND
NOT QRY_TWN_MarchingArmy_PrisonCellShouldBeUsed()
AND
NOT QRY_OncePerUserAndNearbyPlayers_IsSet(_MainCriminal, "MOO_Oubliette_Entry_Once")
THEN
DB_TWN_MarchinArmy_SkipVB(1); // first time we enter Oubliette/Corpse Pit there is a voice bark about it triggered in another place, so we skip ours

PROC
PROC_TWN_MarchingArmy_TeleportCriminalToPrison((CHARACTER)_Criminal)
AND
NOT QRY_TWN_MarchingArmy_PrisonCellShouldBeUsed()
THEN
TeleportTo(_Criminal, S_TWN_MarchingArmy_CorpsePit_9c8e91a2-bf0a-47e6-a38b-d331baa5413f, "", 0);

PROC
PROC_TWN_MarchingArmy_TeleportCriminalToPrison((CHARACTER)_Criminal)
AND
QRY_TWN_MarchingArmy_PrisonCellShouldBeUsed()
AND
DB_RegionPrison("SCL_Main_A_MOO", _PrisonCell, _PrisonSpawnPoint)
AND
DB_PlayerPrison(_PrisonTrigger, _, _, _FugitiveStatus)
THEN
PROC_CrimeTeleportCharacterToPrison(S_MOO_PrisonWarden_66b3e4c0-2f82-4c0a-9333-73a5194f88c7, _Criminal, _PrisonCell, _PrisonSpawnPoint, _FugitiveStatus);

QRY
QRY_TWN_MarchingArmy_PrisonCellShouldBeUsed()
AND
NOT DB_GlobalFlag(MOO_Jailbreak_State_TunnelPathOpen_ef0aa614-2a18-7bd1-0f79-c3aca59ec3ab)
AND
NOT DB_GlobalFlag(MOO_Jailbreak_State_GuardsCleanedOut_d7ca2c67-8eb1-db4d-a16c-2bc3c17deeb7)
THEN
DB_NOOP(1);

IF
EnteredTrigger(_Player, _PrisonTrigger)
AND
DB_TWN_MarchingArmy_PrisonTrigger(_PrisonTrigger)
AND
DB_TWN_MarchingArmy_Criminals(_MainCriminal, _Player)
AND
Random(4000, _Rand)
AND
IntegerSum(_Rand, 1000, _GetUpTimer)
THEN
NOT DB_TWN_MarchingArmy_Criminals(_MainCriminal, _Player);
ApplyStatus(_Player, "KNOCKED_DOWN", 6.0, 1);
RealtimeObjectTimerLaunch(_Player, "TWN_MarchingArmy_AfterCaptured", _GetUpTimer);

IF
ObjectTimerFinished(_Player, "TWN_MarchingArmy_AfterCaptured")
THEN
RemoveStatus(_Player, "KNOCKED_DOWN");

IF
ObjectTimerFinished((CHARACTER)_Player, "TWN_MarchingArmy_AfterCaptured")
AND
DB_TWN_MarchingArmy_MainCriminal(_Player)
AND
NOT DB_TWN_MarchinArmy_SkipVB(1)
AND
QRY_OncePerUserAndNearbyPlayers(_Player, "TWN_MarchingArmy_CommentOnBeingImprisoned")
THEN
StartVoiceBark(TWN_MarchingArmy_VB_Arrested_86eaddd7-ba96-390a-529b-7ec482acd2df, _Player);

IF
ObjectTimerFinished((CHARACTER)_Player, "TWN_MarchingArmy_AfterCaptured")
AND
DB_TWN_MarchingArmy_MainCriminal(_Player)
THEN
NOT DB_TWN_MarchingArmy_MainCriminal(_Player);

IF
ObjectTimerFinished((CHARACTER)_Player, "TWN_MarchingArmy_AfterCaptured")
AND
DB_TWN_MarchinArmy_SkipVB(1)
THEN
NOT DB_TWN_MarchinArmy_SkipVB(1);

// Fallback in case DB_TWN_MarchingArmy_Criminals didn't get cleared
IF
DB_TWN_MarchingArmy_Criminals(_MainCriminal, _OtherCriminal)
AND
NOT DB_TWN_MarchingArmy_MainCriminal(_MainCriminal)
THEN
NOT DB_TWN_MarchingArmy_Criminals(_MainCriminal, _OtherCriminal);
//END_REGION Arrest
