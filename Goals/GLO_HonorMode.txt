Version 1
SubGoalCombiner SGC_AND
INITSECTION

KBSECTION
//REGION Debugcommands
IF
TextEvent("honormode")
AND
CheckRulesetModifierBool(HONOUR_STATS_ef0506df-da9f-40e2-903a-1349523c1ae4,1,_Return)
AND
ConcatenateInteger("Honor mode: ",_Return,_String)
AND
GetHostCharacter(_Player)
THEN
DebugText(_Player,_String);

IF
TextEvent("honormodeon")
AND
DB_CurrentLevel(_LevelName)
THEN
PROC_HonorMode_Enable(_LevelName);

IF
TextEvent("honormodeoff")
AND
DB_CurrentLevel(_LevelName)
THEN
PROC_HonorMode_Disable(_LevelName);
//END_REGION



//REGION Activators
IF
LevelLoaded(_LevelName)
AND
CheckRulesetModifierBool(HONOUR_STATS_ef0506df-da9f-40e2-903a-1349523c1ae4,1,1)
THEN
PROC_HonorMode_Enable(_LevelName);

IF
RulesetModifierChangedBool(HONOUR_STATS_ef0506df-da9f-40e2-903a-1349523c1ae4,0,1)
AND
DB_CurrentLevel(_LevelName)
THEN
PROC_HonorMode_Enable(_LevelName);

IF
LevelLoaded(_LevelName)
AND
CheckRulesetModifierBool(HONOUR_STATS_ef0506df-da9f-40e2-903a-1349523c1ae4,0,1)
THEN
PROC_HonorMode_Disable(_LevelName);

IF
RulesetModifierChangedBool(HONOUR_STATS_ef0506df-da9f-40e2-903a-1349523c1ae4,1,0)
AND
DB_CurrentLevel(_LevelName)
THEN
PROC_HonorMode_Disable(_LevelName);
//END_REGION

PROC
PROC_HonorMode_Enable((STRING)_LevelName)
AND
NOT DB_HonorMode_Enabled(1)
THEN
DB_HonorMode_Enabled(1);

PROC
PROC_HonorMode_Disable((STRING)_LevelName)
AND
DB_HonorMode_Enabled(1)
THEN
NOT DB_HonorMode_Enabled(1);
EXITSECTION

ENDEXITSECTION
