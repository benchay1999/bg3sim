KBSECTION
// Locked Item Tutorial
PROC
PROC_BlockUseOfItem(_Player, _Item)
AND
NOT DB_GLO_Tutorials_LockpickTutorialItem(_Player, _)
AND
DB_Players(_Player)
AND
NOT DB_CustomUseItemResponse(_Player, _Item, 0)
AND
IsLocked(_Item, 1)
THEN
PROC_GLO_Tutorials_TrackedTimerStart(_Player, "GLO_Tutorials_Timer_LockedItemTutorial", 500);
DB_GLO_Tutorials_LockpickTutorialItem((CHARACTER)_Player, _Item);
EXITSECTION

ENDEXITSECTION
