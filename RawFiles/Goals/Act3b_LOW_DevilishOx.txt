//REGION
//Check if the Ox is alive, and if so teleport to the asylum
PROC
PROC_GLO_DevilishOx_TeleportIfAlive()
AND
NOT DB_PermaDefeated(S_GLO_DevilishOx_49eb5b4b-c6ed-4d37-8f68-97b1128b7d56)
THEN
PROC_SetOnStage((CHARACTER)S_GLO_DevilishOx_49eb5b4b-c6ed-4d37-8f68-97b1128b7d56, 1);
TeleportTo(S_GLO_DevilishOx_49eb5b4b-c6ed-4d37-8f68-97b1128b7d56, S_LOW_TeleportDevilishOxPos_Asylum_92e82dd3-f4a9-4c54-b736-efbff958aa65);

PROC_GLO_Origins_SetupRecruitment("CTY_Main_A")
AND
DB_Players(_Player)
AND
QRY_OnlyOnce("WYR_DevilishOx_NoAppleInInventory")
AND
IsInMagicPockets((ITEM)S_GLO_DevilishOx_AppleForm_2fb8359f-7fa7-4dab-8e0f-c1edef958a73, _Player, 0)

AND
QRY_OnlyOnce("WYR_DevilishOx_LeaveNote")
