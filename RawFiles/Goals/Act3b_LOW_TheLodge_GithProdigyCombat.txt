//Make the Mental Image apply a status to the target and viceversa so we can remove the status when one of them dies through stats
IF
StatusApplied(_Char,"LOW_GITHYANKIPRODIGY_MENTALIMAGE_TECHNICAL",_Caster,_)
AND
GetPosition(_Char,_CharX,_CharY,_CharZ)
AND
CreateAt(9b36b1f6-f0a7-4489-ae41-e8aafd58a8ad,_CharX,_CharY,_CharZ,1,1,"",_MentalImage)
THEN
RemoveStatus(_Char,"LOW_GITHYANKIPRODIGY_MENTALIMAGE_TECHNICAL",_MentalImage);
ApplyStatus(_Char,"LOW_GITHYANKIPRODIGY_MENTALIMAGE",-1.0,0,_MentalImage);
ApplyStatus(_MentalImage,"LOW_GITHYANKIPRODIGY_MENTALIMAGE_TECHNICAL_SOURCECHARACTER",-1.0,0,_Char);
ApplyStatus(_MentalImage,"LOW_GITHYANKIPRODIGY_MENTALIMAGE_BEAM",-1.0,0,_Char);

// Hack for no Late Join Penalty during the first round. We don't want Mental Images taking a turn the same round they are created
IF
EnteredCombat(_MentalImage, _CombatGuid)
AND
HasActiveStatus(_MentalImage, "LOW_GITHYANKIPRODIGY_MENTALIMAGE_TECHNICAL_SOURCECHARACTER", 1)
AND
QRY_CurrentCombatRoundIs(_CombatGuid,1)
THEN
ApplyStatus(_MentalImage,"LOW_GITHYANKIPRODIGY_MENTALIMAGE_LATEJOINPENALTY",6.0,0,_MentalImage);

// Poof mental image on destruction
IF
StatusRemoved((CHARACTER)_Char,"LOW_GITHYANKIPRODIGY_MENTALIMAGE_BEAM",_,_)
THEN
PROC_Poof(_Char, VFX_Cinematic_DeathPouch_Poof_01_2bece487-ccbc-9795-40d4-20b64e0e32e5);
