PROC
PROC_EPI_Epilogue_SetupGazettes()
AND
DB_EPI_Epilogue_ElturelSurvivors(_Char)
AND
DB_PermaDefeated(_Char)
THEN
NOT DB_EPI_Epilogue_ElturelSurvivors(_Char);

PROC
PROC_EPI_Epilogue_SetupGazettes()
AND
DB_EPI_Epilogue_FollowUpGazattes(_BookID, _Article)
AND
QRY_EPI_Epilogue_ArticleIsValid(_Article)
THEN
PROC_EPI_Epilogue_SetGazette(_BookID, _Article);

PROC
PROC_EPI_Epilogue_SetGazette((STRING)_BookID, (STRING)_Article)
AND
DB_EPI_Epilogue_FollowUpGazattes(_BookID, _Article)
THEN
AddEntryToCustomBook(_BookID, _Article);

PROC
PROC_EPI_Epilogue_SetGazette((STRING)_BookID, (STRING)_Article)
AND
DB_EPI_Epilogue_FollowUpGazattes(_BookID, _Article)
AND
DB_EPI_Epilogue_FollowUpGazetteImages(_Article, _Image)
THEN
AddIllustrationToCustomBook(_BookID, _Image);

PROC
PROC_EPI_Epilogue_SetGazette((STRING)_BookID, (STRING)_Article)
AND
DB_EPI_Epilogue_FollowUpGazattes(_BookID, _Article)
AND
DB_EPI_Epilogue_FollowUpGazetteWyllImages(_Article, _Image, _IsDevilFlag, _Bool)
AND
GetFlag(_IsDevilFlag, NULL_00000000-0000-0000-0000-000000000000, _Bool)
THEN
AddIllustrationToCustomBook(_BookID, _Image);

PROC
PROC_EPI_Epilogue_SetupGazettes()
AND
DB_EPI_Epilogue_GazetteItems(_BookID, _)
AND
DB_EPI_Epilogue_GazetteEditionNumber(_Num)
AND
Random(10, _Random)
AND
IntegerSum(_Num, _Random, _RandomSum) 
AND
IntegerSum(_RandomSum, 1, _NextNum) //Since random can return 0, which would cause the diplicate entries
THEN
SetEditionForCustomBook(_BookID, _Num);
NOT DB_EPI_Epilogue_GazetteEditionNumber(_Num);
DB_EPI_Epilogue_GazetteEditionNumber(_NextNum);

IF
UseFinished(_Player, _Item, _)
AND
DB_EPI_Epilogue_GazetteItems(_BookID, _Item)
THEN
OpenCustomBookUI(_Player, _BookID);
