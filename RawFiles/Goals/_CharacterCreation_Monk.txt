//REGION Temporary workaround - to set the animation set status when monk character is chosen
PROC
PROC_Shared_CharacterCreationFinished()
THEN
DB_MonkStatus(1);

IF
DB_MonkStatus(1)
AND
DB_CurrentLevel(_Level)
AND
IsGameLevel(_Level, 1)
AND
DB_Players(_Player)
AND
IsTagged(_Player,(TAG)MONK_e1e460bb-d0ae-4452-8529-c9e176558731, 1)
THEN
ApplyStatus(_Player, "MONK_ANIMATION_SET", -1.0, 0, _Player);

IF
DB_MonkStatus(1)
AND
DB_CurrentLevel(_Level)
AND
IsGameLevel(_Level, 1)
THEN
GoalCompleted;
//END_REGION
EXITSECTION
